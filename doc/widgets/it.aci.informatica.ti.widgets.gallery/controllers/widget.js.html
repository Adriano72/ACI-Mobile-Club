exports.removeEventListener = function(eventName, eventHandler) {
    var index = _(eventHandlers[eventName]).indexOf(eventHandler);
    if (index > -1) {
        eventHandlers[eventName].splice(index, 1);
    }
};

function fireEvent(name, params) {

    var self = this;
